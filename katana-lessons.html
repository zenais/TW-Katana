Get match by pattern

You are given a string with three lowercase letters ( pattern ).

Your Task

Implement a function find_matched_by_pattern(pattern) that returns a predicate function, testing a string input and returning true if the string is matching the pattern, false otherwise. A word is considered a match for a given pattern if the first occurrence of each letter of the pattern is found in the same order in the word. If a character in the pattern is duplicated, the same logic applies further. The pattern will always be a string of size 3. 
    function findMatchedByPattern(pattern) {
        // TODO: implement this!
        const f = (e) => {
            //
            let C1 = e.indexOf(pattern[0]);
        
            let C2 = e.indexOf(pattern[1]);
            if (C2 < C1) return false;
            C2 = e.substring(C1+1).indexOf(pattern[1]);
        
            let C3 = e.indexOf(pattern[2]);
            if (C3 < C1) return false;
            C3 = e.substring(C1+1).indexOf(pattern[2]);
            if (C3 < C2) return false;
            C3 = e.substring(C2 + C1 + 2).indexOf(pattern[2]);
        
            return C1 >= 0 && C2 >= 0 && C3 >= 0;
        };
        return f;
    }
